Configures TIMER2 for basic operation. Sets up default prescaler and period values. Actual PWM frequency is configured via \doxylink{group___t_i_m_e_r___f_u_n_c_t_i_o_n_s_gae1f2c21be6c10594144a9665d427913a}{Timer\+Set\+Frequency()}.

{\bfseries{TIMER2 Hardware:\+}}
\begin{DoxyItemize}
\item Type:\+ 32-\/bit general-\/purpose timer
\item Clock:\+ APB1 (42 MHz after RCC prescaler)
\item Channels:\+ 4 PWM outputs (CH1-\/\+CH4)
\item Prescaler Range:\+ 1-\/65536
\item Period Range:\+ 1-\/0x\+FFFFFFFF (32-\/bit)
\end{DoxyItemize}

{\bfseries{Register Configuration:\+}}
\begin{DoxyItemize}
\item TIM\+\_\+\+PSC:\+ Prescaler value
\item TIM\+\_\+\+ARR:\+ Auto-\/reload register (period)
\item TIM\+\_\+\+CR1:\+ Control register (enable, direction)
\item TIM\+\_\+\+CCMRx:\+ Capture/\+compare mode (PWM mode 1)
\item TIM\+\_\+\+CCER:\+ Capture/\+compare enable
\item TIM\+\_\+\+CCRx:\+ Capture/\+compare register (duty cycle)
\end{DoxyItemize}

{\bfseries{Timer Modes:\+}}
\begin{DoxyItemize}
\item Up-\/\+Counter:\+ Counts 0 → ARR → 0 (edge-\/aligned PWM)
\item Down-\/\+Counter:\+ Counts ARR → 0 → ARR (alternative timing)
\end{DoxyItemize}
