Receives data through USART with blocking mode. Function waits for receive register to have data (RXNE flag) before reading each byte. Blocks until all requested data is received.

\begin{DoxyCode}{0}
\DoxyCodeLine{For\ each\ \textcolor{keywordtype}{byte}\ to\ receive:}

\DoxyCodeLine{\ \ 2.\ Read\ \textcolor{keywordtype}{byte}\ from\ RDR}
\DoxyCodeLine{\ \ 3.\ Process\ based\ on\ word\ length\ and\ parity}
\DoxyCodeLine{\ \ 4.\ Store\ in\ \doxymbox{\hyperlink{main_8c_a5819dbc2d305e99e930c734a0d28bc3b}{buffer}}}

{\bfseries{Word Length and Parity Handling:\+}}
\begin{DoxyItemize}
\item 8-\/bit, no parity:\+ Read full 8 bits, store 1 byte
\item 8-\/bit, with parity:\+ Read 7 data bits + parity, store 1 byte (0x7F mask)
\item 9-\/bit, no parity:\+ Read full 9 bits, store 2 bytes (0x1\+FF mask)
\item 9-\/bit, with parity:\+ Read 8 data bits + parity, store 1 byte (0x\+FF mask)
\end{DoxyItemize}

{\bfseries{Masking Details:\+}}
\begin{DoxyItemize}
\item Masks remove unused bits from RDR (hardware may set extra bits)
\item 8-\/bit no parity:\+ mask 0x\+FF (all 8 bits)
\item 8-\/bit with parity:\+ mask 0x7F (7 data bits, drop parity bit)
\item 9-\/bit no parity:\+ mask 0x1\+FF (all 9 bits)
\item 9-\/bit with parity:\+ mask 0x\+FF (8 data bits, drop parity bit)
\end{DoxyItemize}

\begin{DoxyWarning}{Warning}

\begin{DoxyItemize}
\item BRR value of 0 or 1 causes USART to malfunction
\item Very high baud rates (\texorpdfstring{$>$}{>} 1Mbps) may need different clock source
\item Clock frequency changes require recalculation
\item Terminal must match baud rate exactly
\end{DoxyItemize}
\end{DoxyWarning}
